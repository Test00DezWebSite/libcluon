43  std::string dataToSend;
44  {
45  cluon::ToProtoVisitor protoEncoder;
46  envelope.accept(protoEncoder);
47 
48  const std::string tmp{protoEncoder.encodedData()};
49  uint32_t length{static_cast<uint32_t>(tmp.size())};
50  length = htole32(length);
51 
52  // Add OpenDaVINCI header.
53  std::array<char, 5> header;
54  header[0] = static_cast<char>(0x0D);
55  header[1] = static_cast<char>(0xA4);
56  header[2] = *(reinterpret_cast<char *>(&length) + 0);

58  header[4] = *(reinterpret_cast<char *>(&length) + 2);
59

64  }
